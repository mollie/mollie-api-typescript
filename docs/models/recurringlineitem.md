# RecurringLineItem

## Example Usage

```typescript
import { RecurringLineItem } from "mollie-api-typescript/models";

let value: RecurringLineItem = {
  description: "Gym subscription",
  interval: "12 months",
  amount: {
    currency: "EUR",
    value: "10.00",
  },
  times: 1,
  startDate: "2024-12-12",
};
```
