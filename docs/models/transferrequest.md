# TransferRequest

## Example Usage

```typescript
import { TransferRequest } from "mollie-api-typescript/models";

let value: TransferRequest = {
  debtorIban: "NL55MLLE0123456789",
  creditor: {
    fullName: "Jan Jansen",
    account: {
      iban: "NL02ABNA0123456789",
    },
  },
  amount: {
    currency: "EUR",
    value: "10.00",
  },
  description: "Invoice 12345",
  transferScheme: {
    type: "sepa-credit-inst",
  },
  testmode: false,
};
```
