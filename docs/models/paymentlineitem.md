# PaymentLineItem

## Example Usage

```typescript
import { PaymentLineItem } from "mollie-api-typescript/models";

let value: PaymentLineItem = {
  type: "physical",
  description: "LEGO 4440 Forest Police Station",
  quantity: 1,
  quantityUnit: "pcs",
  unitPrice: {
    currency: "EUR",
    value: "10.00",
  },
  discountAmount: {
    currency: "EUR",
    value: "10.00",
  },
  totalAmount: {
    currency: "EUR",
    value: "10.00",
  },
  vatRate: "21.00",
  vatAmount: {
    currency: "EUR",
    value: "10.00",
  },
  sku: "9780241661628",
  categories: [
    "meal",
    "eco",
  ],
  imageUrl: "https://...",
  productUrl: "https://...",
};
```
