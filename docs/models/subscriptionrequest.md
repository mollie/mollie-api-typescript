# SubscriptionRequest

## Example Usage

```typescript
import { SubscriptionRequest } from "mollie-api-typescript/models";

let value: SubscriptionRequest = {
  amount: {
    currency: "EUR",
    value: "10.00",
  },
  times: 6,
  interval: "2 days",
  startDate: "2025-01-01",
  description: "Subscription of streaming channel",
  method: "paypal",
  applicationFee: {
    amount: {
      currency: "EUR",
      value: "10.00",
    },
    description: "Platform fee",
  },
  webhookUrl: "https://example.com/webhook",
  mandateId: "mdt_5B8cwPMGnU",
  profileId: "pfl_5B8cwPMGnU",
  testmode: false,
};
```
